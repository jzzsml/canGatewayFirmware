79 static void can_SetBaudrate (LPC_CAN_TypeDef *CANx, uint32_t baudrate)
80 {
81  uint32_t result = 0;
82  uint8_t NT, TSEG1, TSEG2, BRFail;
83  uint32_t CANPclk = 0;
84  uint32_t BRP;
85  CHECK_PARAM(PARAM_CANx(CANx));
86 
87  if (CANx == LPC_CAN1)
88  {
89  CANPclk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_CAN1);
90  }
91  else
92  {
93  CANPclk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_CAN2);
94  }
95  result = CANPclk / baudrate;
96  /* Calculate suitable nominal time value
97  * NT (nominal time) = (TSEG1 + TSEG2 + 3)
98  * NT <= 24
99  * TSEG1 >= 2*TSEG2
100  */
101  BRFail = 1;
102  for(NT=24;NT>0;NT=NT-2)
103  {
104  if ((result%NT)==0)
105  {
106  BRP = result / NT - 1;
107  NT--;
108  TSEG2 = (NT/3) - 1;
109  TSEG1 = NT -(NT/3) - 1;
110  BRFail = 0;
111  break;
112  }
113  }
114  if(BRFail)
115  while(1); // Failed to calculate exact CAN baud rate
116  /* Enter reset mode */
117  CANx->MOD = 0x01;
118  /* Set bit timing
119  * Default: SAM = 0x00;
120  * SJW = 0x03;
121  */
122  CANx->BTR = (TSEG2<<20)|(TSEG1<<16)|(3<<14)|BRP;
123  /* Return to normal operating */
124  CANx->MOD = 0;
125 }

496 CAN_ERROR CAN_LoadExplicitEntry(LPC_CAN_TypeDef* CANx, uint32_t id, CAN_ID_FORMAT_Type format)
497 {
498  uint32_t tmp0 = 0;
499  uint32_t buf0=0, buf1=0;
500  int16_t cnt1=0, cnt2=0, bound1=0, total=0;
501 
502 
503  CHECK_PARAM(PARAM_CANx(CANx));
504  CHECK_PARAM(PARAM_ID_FORMAT(format));
505 
506  if (CANx == LPC_CAN1)
507  {
508  tmp0 = 0;
509  }
510  else if (CANx == LPC_CAN2)
511  {
512  tmp0 = 1;
513  }
514 
515  /* Acceptance Filter Memory full - return */
516  total =((CANAF_FullCAN_cnt+1)>>1)+ CANAF_FullCAN_cnt*3 +((CANAF_std_cnt + 1) >> 1)+ \
517  CANAF_gstd_cnt + CANAF_ext_cnt + (CANAF_gext_cnt<<1);
518  if (total >= 512){ //don't have enough space
519  return CAN_OBJECTS_FULL_ERROR;
520  }
521 
522  /* Setup Acceptance Filter Configuration
523  Acceptance Filter Mode Register = Off */
524  LPC_CANAF->AFMR = 0x00000001;
525 
526 /*********** Add Explicit Standard Identifier Frame Format entry *********/
527  if(format == STD_ID_FORMAT)
528  {
529  id &= 0x07FF;
530  id |= (tmp0 << 13); /* Add controller number */
531  /* Move all remaining sections one place up
532  if new entry will increase FullCAN list */
533  if ((CANAF_std_cnt & 0x0001) == 0)
534  {
535  cnt1 = ((CANAF_FullCAN_cnt+1)>>1)+((CANAF_std_cnt+1)>>1);
536  bound1 = total - cnt1;
537  buf0 = LPC_CANAF_RAM->mask[cnt1];
538  while(bound1--)
539  {
540  cnt1++;
541  buf1 = LPC_CANAF_RAM->mask[cnt1];
542  LPC_CANAF_RAM->mask[cnt1] = buf0;
543  buf0 = buf1;
544  }
545  }
546  if (CANAF_std_cnt == 0)
547  {
548  cnt2 = (CANAF_FullCAN_cnt + 1)>>1;
549  /* For entering first ID */
550  LPC_CANAF_RAM->mask[cnt2] = 0x0000FFFF | (id << 16);
551  }
552  else if (CANAF_std_cnt == 1)
553  {
554  cnt2 = (CANAF_FullCAN_cnt + 1)>>1;
555  /* For entering second ID */
556  if (((LPC_CANAF_RAM->mask[cnt2] >> 16)& 0xE7FF) > id)
557  {
558  LPC_CANAF_RAM->mask[cnt2] = (LPC_CANAF_RAM->mask[cnt2] >> 16) | (id << 16);
559  }
560  else
561  {
562  LPC_CANAF_RAM->mask[cnt2] = (LPC_CANAF_RAM->mask[cnt2] & 0xFFFF0000) | id;
563  }
564  }
565  else
566  {
567  /* Find where to insert new ID */
568  cnt1 = (CANAF_FullCAN_cnt+1)>>1;
569  cnt2 = CANAF_std_cnt;
570  bound1 = ((CANAF_FullCAN_cnt+1)>>1)+((CANAF_std_cnt+1)>>1);
571  while (cnt1 < bound1)
572  {
573  /* Loop through standard existing IDs */
574  if (((LPC_CANAF_RAM->mask[cnt1] >> 16) & 0xE7FF) > id)
575  {
576  cnt2 = cnt1 * 2;
577  break;
578  }
579 
580  if ((LPC_CANAF_RAM->mask[cnt1] & 0x0000E7FF) > id)
581  {
582  cnt2 = cnt1 * 2 + 1;
583  break;
584  }
585 
586  cnt1++;
587  }
588  /* cnt1 = U32 where to insert new ID */
589  /* cnt2 = U16 where to insert new ID */
590 
591  if (cnt1 == bound1)
592  {
593  /* Adding ID as last entry */
594  /* Even number of IDs exists */
595  if ((CANAF_std_cnt & 0x0001) == 0)
596  {
597  LPC_CANAF_RAM->mask[cnt1] = 0x0000FFFF | (id << 16);
598  }
599  /* Odd number of IDs exists */
600  else
601  {
602  LPC_CANAF_RAM->mask[cnt1] = (LPC_CANAF_RAM->mask[cnt1] & 0xFFFF0000) | id;
603  }
604  }
605  else
606  {
607  buf0 = LPC_CANAF_RAM->mask[cnt1]; /* Remember current entry */
608  if ((cnt2 & 0x0001) == 0)
609  {
610  /* Insert new mask to even address*/
611  buf1 = (id << 16) | (buf0 >> 16);
612  }
613  else
614  {
615  /* Insert new mask to odd address */
616  buf1 = (buf0 & 0xFFFF0000) | id;
617  }
618  LPC_CANAF_RAM->mask[cnt1] = buf1;/* Insert mask */
619  bound1 = ((CANAF_FullCAN_cnt+1)>>1)+((CANAF_std_cnt+1)>>1)-1;
620  /* Move all remaining standard mask entries one place up */
621  while (cnt1 < bound1)
622  {
623  cnt1++;
624  buf1 = LPC_CANAF_RAM->mask[cnt1];
625  LPC_CANAF_RAM->mask[cnt1] = (buf1 >> 16) | (buf0 << 16);
626  buf0 = buf1;
627  }
628 
629  if ((CANAF_std_cnt & 0x0001) == 0)
630  {
631  /* Even number of IDs exists */
632  LPC_CANAF_RAM->mask[cnt1+1] = (buf0 <<16) |(0x0000FFFF);
633  }
634  }
635  }
636  CANAF_std_cnt++;
637  //update address values
638  LPC_CANAF->SFF_GRP_sa +=0x04 ;
639  LPC_CANAF->EFF_sa +=0x04 ;
640  LPC_CANAF->EFF_GRP_sa +=0x04;
641  LPC_CANAF->ENDofTable +=0x04;
642  }
643 
644 /*********** Add Explicit Extended Identifier Frame Format entry *********/
645  else
646  {
647  /* Add controller number */
648  id |= (tmp0) << 29;
649 
650  cnt1 = ((CANAF_FullCAN_cnt+1)>>1)+(((CANAF_std_cnt + 1) >> 1) + CANAF_gstd_cnt);
651  cnt2 = 0;
652  while (cnt2 < CANAF_ext_cnt)
653  {
654  /* Loop through extended existing masks*/
655  if (LPC_CANAF_RAM->mask[cnt1] > id)
656  {
657  break;
658  }
659  cnt1++;/* cnt1 = U32 where to insert new mask */
660  cnt2++;
661  }
662 
663  buf0 = LPC_CANAF_RAM->mask[cnt1]; /* Remember current entry */
664  LPC_CANAF_RAM->mask[cnt1] = id; /* Insert mask */
665 
666  CANAF_ext_cnt++;
667 
668  bound1 = total;
669  /* Move all remaining extended mask entries one place up*/
670  while (cnt2 < bound1)
671  {
672  cnt1++;
673  cnt2++;
674  buf1 = LPC_CANAF_RAM->mask[cnt1];
675  LPC_CANAF_RAM->mask[cnt1] = buf0;
676  buf0 = buf1;
677  }
678  /* update address values */
679  LPC_CANAF->EFF_GRP_sa += 4;
680  LPC_CANAF->ENDofTable += 4;
681  }
682  if(CANAF_FullCAN_cnt == 0) //not use FullCAN mode
683  {
684  LPC_CANAF->AFMR = 0x00;//not use FullCAN mode
685  }
686  else
687  {
688  LPC_CANAF->AFMR = 0x04;
689  }
690 
691  return CAN_OK;
692 }
