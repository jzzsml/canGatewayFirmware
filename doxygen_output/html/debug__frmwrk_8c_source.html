184 void UARTPutHex (LPC_UART_TypeDef *UARTx, uint8_t hexnum)
185 {
186  uint8_t nibble, i;
187 
188  UARTPuts(UARTx, "0x");
189  i = 1;
190  do {
191  nibble = (hexnum >> (4*i)) & 0x0F;
192  UARTPutChar(UARTx, (nibble > 9) ? ('A' + nibble - 10) : ('0' + nibble));
193  } while (i--);
194 }
