66 static Status uart_set_divisors(LPC_UART_TypeDef *UARTx, uint32_t baudrate)
67 {
68  Status errorStatus = ERROR;
69 
70  uint32_t uClk;
71  uint32_t d, m, bestd, bestm, tmp;
72  uint64_t best_divisor, divisor;
73  uint32_t current_error, best_error;
74  uint32_t recalcbaud;
75 
76  /* get UART block clock */
77  if (UARTx == LPC_UART0)
78  {
79  uClk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_UART0);
80  }
81  else if (UARTx == (LPC_UART_TypeDef *)LPC_UART1)
82  {
83  uClk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_UART1);
84  }
85  else if (UARTx == LPC_UART2)
86  {
87  uClk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_UART2);
88  }
89  else if (UARTx == LPC_UART3)
90  {
91  uClk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_UART3);
92  }
93 
94 
95  uClk = uClk >> 4; /* div by 16 */
96  /* In the Uart IP block, baud rate is calculated using FDR and DLL-DLM registers
97  * The formula is :
98  * BaudRate= uClk * (mulFracDiv/(mulFracDiv+dividerAddFracDiv) / (16 * (DLL)
99  * It involves floating point calculations. That's the reason the formulae are adjusted with
100  * Multiply and divide method.*/
101  /* The value of mulFracDiv and dividerAddFracDiv should comply to the following expressions:
102  * 0 < mulFracDiv <= 15, 0 <= dividerAddFracDiv <= 15 */
103  best_error = 0xFFFFFFFF; /* Worst case */
104  bestd = 0;
105  bestm = 0;
106  best_divisor = 0;
107  for (m = 1 ; m <= 15 ;m++)
108  {
109  for (d = 0 ; d < m ; d++)
110  {
111  divisor = ((uint64_t)uClk<<28)*m/(baudrate*(m+d));
112  current_error = divisor & 0xFFFFFFFF;
113 
114  tmp = divisor>>32;
115 
116  /* Adjust error */
117  if(current_error > ((uint32_t)1<<31)){
118  current_error = -current_error;
119  tmp++;
120  }
121 
122  if(tmp<1 || tmp>65536) /* Out of range */
123  continue;
124 
125  if( current_error < best_error){
126  best_error = current_error;
127  best_divisor = tmp;
128  bestd = d;
129  bestm = m;
130  if(best_error == 0) break;
131  }
132  } /* end of inner for loop */
133 
134  if (best_error == 0)
135  break;
136  } /* end of outer for loop */
137 
138  if(best_divisor == 0) return ERROR; /* can not find best match */
139 
140  recalcbaud = (uClk>>4) * bestm/(best_divisor * (bestm + bestd));
141 
142  /* reuse best_error to evaluate baud error*/
143  if(baudrate>recalcbaud) best_error = baudrate - recalcbaud;
144  else best_error = recalcbaud -baudrate;
145 
146  best_error = best_error * 100 / baudrate;
147 
148  if (best_error < UART_ACCEPTED_BAUDRATE_ERROR)
149  {
150  if (((LPC_UART1_TypeDef *)UARTx) == LPC_UART1)
151  {
152  ((LPC_UART1_TypeDef *)UARTx)->LCR |= UART_LCR_DLAB_EN;
153  ((LPC_UART1_TypeDef *)UARTx)->/*DLIER.*/DLM = UART_LOAD_DLM(best_divisor);
154  ((LPC_UART1_TypeDef *)UARTx)->/*RBTHDLR.*/DLL = UART_LOAD_DLL(best_divisor);
155  /* Then reset DLAB bit */
156  ((LPC_UART1_TypeDef *)UARTx)->LCR &= (~UART_LCR_DLAB_EN) & UART_LCR_BITMASK;
157  ((LPC_UART1_TypeDef *)UARTx)->FDR = (UART_FDR_MULVAL(bestm) \
158  | UART_FDR_DIVADDVAL(bestd)) & UART_FDR_BITMASK;
159  }
160  else
161  {
162  UARTx->LCR |= UART_LCR_DLAB_EN;
163  UARTx->/*DLIER.*/DLM = UART_LOAD_DLM(best_divisor);
164  UARTx->/*RBTHDLR.*/DLL = UART_LOAD_DLL(best_divisor);
165  /* Then reset DLAB bit */
166  UARTx->LCR &= (~UART_LCR_DLAB_EN) & UART_LCR_BITMASK;
167  UARTx->FDR = (UART_FDR_MULVAL(bestm) \
168  | UART_FDR_DIVADDVAL(bestd)) & UART_FDR_BITMASK;
169  }
170  errorStatus = SUCCESS;
171  }
172 
173  return errorStatus;
174 }

527 uint32_t UART_Send(LPC_UART_TypeDef *UARTx, uint8_t *txbuf,
528  uint32_t buflen, TRANSFER_BLOCK_Type flag)
529 {
530  uint32_t bToSend, bSent, timeOut, fifo_cnt;
531  uint8_t *pChar = txbuf;
532 
533  bToSend = buflen;
534 
535  // blocking mode
536  if (flag == BLOCKING) {
537  bSent = 0;
538  while (bToSend){
539  timeOut = UART_BLOCKING_TIMEOUT;
540  // Wait for THR empty with timeout
541  while (!(UARTx->LSR & UART_LSR_THRE)) {
542  if (timeOut == 0) break;
543  timeOut--;
544  }
545  // Time out!
546  if(timeOut == 0) break;
547  fifo_cnt = UART_TX_FIFO_SIZE;
548  while (fifo_cnt && bToSend){
549  UART_SendByte(UARTx, (*pChar++));
550  fifo_cnt--;
551  bToSend--;
552  bSent++;
553  }
554  }
555  }
556  // None blocking mode
557  else {
558  bSent = 0;
559  while (bToSend) {
560  if (!(UARTx->LSR & UART_LSR_THRE)){
561  break;
562  }
563  fifo_cnt = UART_TX_FIFO_SIZE;
564  while (fifo_cnt && bToSend) {
565  UART_SendByte(UARTx, (*pChar++));
566  bToSend--;
567  fifo_cnt--;
568  bSent++;
569  }
570  }
571  }
572  return bSent;
573 }

593  uint32_t buflen, TRANSFER_BLOCK_Type flag)
594 {

615  }
616  }

619  bRecv = 0;

621  if (!(UARTx->LSR & UART_LSR_RDR)) {

623  } else {

627  }
628  }
629  }
